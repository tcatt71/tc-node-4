Object Constructors


                        let game = {
                            home: undefined,
                            away: undefined,
                            score: {
                                home: 0,
                                away: 0,
                            },
                            playGame: function () {
                                console.log(
                                    `The ${this.away.city} ${this.away.name} is playing at the ${this.home.city} ${this.home.name}...`
                                );
                            },
                        };
                        
                        function Game(home, away) {
                            let obj = Object.create(game);
                            obj.home = home;
                            obj.away = away;
                            return obj;
                        }

                        let game1 = new Game(trailblazers, 76ers);
